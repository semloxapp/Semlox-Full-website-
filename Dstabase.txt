-- Member roles
DO $$
BEGIN
  IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'member_role') THEN
    CREATE TYPE member_role AS ENUM ('owner', 'admin', 'manager', 'operator', 'viewer');
  END IF;
END$$;

-- Document lifecycle status
DO $$
BEGIN
  IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'document_status') THEN
    CREATE TYPE document_status AS ENUM (
      'uploaded', 'processing', 'extracted', 'reviewed', 'issued', 'failed'
    );
  END IF;
END$$;

-- AWB status
DO $$
BEGIN
  IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'awb_status') THEN
    CREATE TYPE awb_status AS ENUM ('draft', 'issued', 'cancelled');
  END IF;
END$$;

-- AWB number status
DO $$
BEGIN
  IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'awb_number_status') THEN
    CREATE TYPE awb_number_status AS ENUM ('available', 'reserved', 'consumed', 'cancelled');
  END IF;
END$$;

-- Audit action type
DO $$
BEGIN
  IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'audit_action') THEN
    CREATE TYPE audit_action AS ENUM (
      'create', 'update', 'delete', 'login', 'logout', 'export', 'issue', 'approve'
    );
  END IF;
END$$;

-- Helper function: updated_at trigger (no dependencies)
CREATE OR REPLACE FUNCTION set_updated_at()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN
  NEW.updated_at := NOW();
  RETURN NEW;
END$$;

-- Check company membership
CREATE OR REPLACE FUNCTION auth_is_member(p_company UUID)
RETURNS BOOLEAN LANGUAGE SQL STABLE AS $$
  SELECT EXISTS (
    SELECT 1 FROM memberships m
    WHERE m.company_id = p_company AND m.user_id = auth.uid()
  );
$$;

-- Check if user is owner/admin
CREATE OR REPLACE FUNCTION auth_is_admin(p_company UUID)
RETURNS BOOLEAN LANGUAGE SQL STABLE AS $$
  SELECT EXISTS (
    SELECT 1 FROM memberships m
    WHERE m.company_id = p_company
      AND m.user_id = auth.uid()
      AND m.role IN ('owner', 'admin')
  );
$$;

-- Get AWB stock for a company
CREATE OR REPLACE FUNCTION get_awb_stock(p_company UUID)
RETURNS TABLE(total BIGINT, available BIGINT, consumed BIGINT, reserved BIGINT) 
LANGUAGE SQL STABLE AS $$
  SELECT
    COUNT(*)::BIGINT,
    COUNT(*) FILTER (WHERE status = 'available')::BIGINT,
    COUNT(*) FILTER (WHERE status = 'consumed')::BIGINT,
    COUNT(*) FILTER (WHERE status = 'reserved')::BIGINT
  FROM awb_numbers
  WHERE company_id = p_company;
$$;

-- Consume next available AWB number
CREATE OR REPLACE FUNCTION consume_next_awb_number(p_company UUID)
RETURNS JSONB LANGUAGE plpgsql SECURITY DEFINER AS $$
DECLARE
  r awb_numbers%ROWTYPE;
BEGIN
  SELECT * INTO r
  FROM awb_numbers
  WHERE company_id = p_company AND status = 'available'
  ORDER BY added_at ASC
  FOR UPDATE SKIP LOCKED
  LIMIT 1;

  IF NOT FOUND THEN
    RETURN jsonb_build_object(
      'success', FALSE,
      'awb_number', NULL,
      'error', 'No available AWB numbers'
    );
  END IF;

  UPDATE awb_numbers
  SET status = 'consumed', consumed_at = NOW(), consumed_by = auth.uid()
  WHERE id = r.id;

  RETURN jsonb_build_object(
    'success', TRUE,
    'awb_number', r.awb_number,
    'error', NULL
  );
END$$;

-- Atomic: consume next available AWB number for a company
CREATE OR REPLACE FUNCTION consume_next_awb_number(p_company UUID)
RETURNS JSONB LANGUAGE plpgsql SECURITY DEFINER AS $$
DECLARE
  r awb_numbers%ROWTYPE;
BEGIN
  SELECT * INTO r
  FROM awb_numbers
  WHERE company_id = p_company AND status = 'available'
  ORDER BY added_at ASC
  FOR UPDATE SKIP LOCKED
  LIMIT 1;

  IF NOT FOUND THEN
    RETURN jsonb_build_object(
      'success', FALSE,
      'awb_number', NULL,
      'error', 'No available AWB numbers in inventory'
    );
  END IF;

  UPDATE awb_numbers
  SET status = 'consumed', consumed_at = NOW(), consumed_by = auth.uid()
  WHERE id = r.id;

  RETURN jsonb_build_object(
    'success', TRUE,
    'awb_number', r.awb_number,
    'error', NULL
  );
END$$;

-- Get remaining AWB stock for a company
CREATE OR REPLACE FUNCTION get_awb_stock(p_company UUID)
RETURNS TABLE(total BIGINT, available BIGINT, consumed BIGINT, reserved BIGINT) 
LANGUAGE SQL STABLE AS $$
  SELECT
    COUNT(*)::BIGINT AS total,
    COUNT(*) FILTER (WHERE status = 'available')::BIGINT AS available,
    COUNT(*) FILTER (WHERE status = 'consumed')::BIGINT AS consumed,
    COUNT(*) FILTER (WHERE status = 'reserved')::BIGINT AS reserved
  FROM awb_numbers
  WHERE company_id = p_company;
$$;

-- ======================================================
-- PHASE 12.5: MEMBERSHIP-DEPENDENT FUNCTIONS (After memberships table exists)
-- ======================================================

-- Check company membership (used in RLS policies)
CREATE OR REPLACE FUNCTION auth_is_member(p_company UUID)
RETURNS BOOLEAN LANGUAGE SQL STABLE AS $$
  SELECT EXISTS (
    SELECT 1 FROM memberships m
    WHERE m.company_id = p_company AND m.user_id = auth.uid()
  );
$$;

-- Check if user is company owner/admin
CREATE OR REPLACE FUNCTION auth_is_admin(p_company UUID)
RETURNS BOOLEAN LANGUAGE SQL STABLE AS $$
  SELECT EXISTS (
    SELECT 1 FROM memberships m
    WHERE m.company_id = p_company
      AND m.user_id = auth.uid()
      AND m.role IN ('owner', 'admin')
  );
$$;
